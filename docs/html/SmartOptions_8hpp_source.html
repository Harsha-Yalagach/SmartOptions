333  if ('-' == token[0])
334  {

347

354  // Second Process Options...
355  if (false == isTokenProcessed) {
356  for (SmartOptionsOptionArgList::iterator optionsIt = this->options.begin();
357  optionsIt != this->options.end();
358  optionsIt++)
359  {
360  if ( (*optionsIt).prefixShort == (*(char*)token) ) {
361  SmartOptionsOptionArg optionArg = (SmartOptionsOptionArg)(*optionsIt);
362 
363  // Update the variable that has been passed while configuring...
364  if (SmartOptions::NULL_TERMINATE != token[1]) {
365  // If the argument provided is not separated by space...
366  *(optionArg.destVariable) = ++token;
367  isTokenProcessed = true;
368  }
369  else if (index >= (argc-1)) {
370  strErrMessage = std::string(": Error, missing value for '-") + (*optionsIt).prefixShort + "' option.";
371  } else {
372  // If the argument provided is separated by space...
373  const char *optionStr = this->argV[++index];
374  *(optionArg.destVariable) = optionStr;
375  isTokenProcessed = true;
376  }
377  }
378 
379  if (isTokenProcessed) break;
380  }
381  }
382 
383  // Flag error, if the token is not processed...
384  if (false == isTokenProcessed) {
385  if (this->autoPrintHelp) {
386  if (strErrMessage.empty() == false)
387  std::cout << std::string(this->appName) << strErrMessage << std::endl;
388  else
389  std::cout << std::string(this->appName) << ": Error, invalid argument '-" << token[0] << "'." << std::endl;
390  }

393  }
